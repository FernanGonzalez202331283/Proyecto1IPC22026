CREATE DATABASE IF NOT EXISTS agencia;
USE agencia;

CREATE TABLE usuario (
    id INT AUTO_INCREMENT PRIMARY KEY,
    username VARCHAR(50) NOT NULL,
    password VARCHAR(100) NOT NULL,
    rol ENUM('ADMIN', 'OPERACIONES', 'ATENCION') NOT NULL,
    estado TINYINT(1) DEFAULT 1
);

CREATE TABLE cliente (
    dpi VARCHAR(20) PRIMARY KEY,
    nombre VARCHAR(100),
    fecha_nacimiento DATE,
    telefono VARCHAR(20),
    correo VARCHAR(100),
    nacionalidad VARCHAR(50)
);

CREATE TABLE destino (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(100) NOT NULL UNIQUE,
    pais VARCHAR(100) NOT NULL,
    descripcion TEXT NOT NULL,
    clima VARCHAR(100),
    imagen_url TEXT
);

CREATE TABLE proveedor (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(100) NOT NULL UNIQUE,
    tipo VARCHAR(50) NOT NULL,
    pais VARCHAR(100) NOT NULL,
    contacto VARCHAR(100)
);

CREATE TABLE paquete (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(100) NOT NULL UNIQUE,
    destino_id INT NOT NULL,
    duracion INT NOT NULL,
    descripcion TEXT,
    precio DOUBLE NOT NULL,
    capacidad INT NOT NULL,
    estado ENUM('activo', 'inactivo') NOT NULL,
    CONSTRAINT fk_paquete_destino FOREIGN KEY (destino_id) REFERENCES destino(id)
);

CREATE TABLE servicio (
    id INT AUTO_INCREMENT PRIMARY KEY,
    paquete_id INT NOT NULL,
    proveedor_id INT NOT NULL,
    nombre VARCHAR(100),
    costo DOUBLE,
    CONSTRAINT fk_servicio_paquete FOREIGN KEY (paquete_id) REFERENCES paquete(id),
    CONSTRAINT fk_servicio_proveedor FOREIGN KEY (proveedor_id) REFERENCES proveedor(id)
);

CREATE TABLE reservacion (
    id INT AUTO_INCREMENT PRIMARY KEY,
    fecha_creacion DATETIME DEFAULT CURRENT_TIMESTAMP,
    fecha_viaje DATE,
    cantidad_personas INT,
    costo_total DECIMAL(10, 2),
    estado ENUM('PENDIENTE', 'CONFIRMADA', 'CANCELADA', 'COMPLETADA') DEFAULT 'PENDIENTE',
    paquete_id INT,
    usuario_id INT,
    CONSTRAINT fk_reserva_paquete FOREIGN KEY (paquete_id) REFERENCES paquete(id),
    CONSTRAINT fk_reserva_usuario FOREIGN KEY (usuario_id) REFERENCES usuario(id)
);

CREATE TABLE reservacion_cliente (
    id INT AUTO_INCREMENT PRIMARY KEY,
    reservacion_id INT,
    cliente_dpi VARCHAR(20),
    CONSTRAINT fk_rc_reservacion FOREIGN KEY (reservacion_id) REFERENCES reservacion(id),
    CONSTRAINT fk_rc_cliente FOREIGN KEY (cliente_dpi) REFERENCES cliente(dpi)
);

CREATE TABLE pago (
    id INT AUTO_INCREMENT PRIMARY KEY,
    reservacion_id INT,
    monto DECIMAL(10, 2),
    metodo VARCHAR(50),
    fecha DATETIME DEFAULT CURRENT_TIMESTAMP,
    CONSTRAINT fk_pago_reservacion FOREIGN KEY (reservacion_id) REFERENCES reservacion(id)
);

CREATE TABLE cancelacion (
    id INT AUTO_INCREMENT PRIMARY KEY,
    reservacion_id INT,
    fecha DATETIME DEFAULT CURRENT_TIMESTAMP,
    monto_reembolso DECIMAL(10, 2),
    perdida DECIMAL(10, 2),
    CONSTRAINT fk_cancel_reservacion FOREIGN KEY (reservacion_id) REFERENCES reservacion(id)
);
